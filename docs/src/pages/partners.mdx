export function PartnerCard({
    icon,
    banner,
    name,
    description,
    color,
    links,
    isDiscordServer
}) {
    return (
        <div className="w-full rounded-lg overflow-hidden bg-white dark:bg-neutral-900 relative border dark:border-0 border-neutral-200 drop-shadow-md">
            {banner ? (
                <div className="bg-neutral-700 h-32">
                    <img src={banner} className="h-32 w-full object-cover" />
                </div>
            ) : (
                <div
                    className="h-32 bg-neutral-100 dark:bg-neutral-800"
                    style={{
                        background: `radial-gradient(${color} 0.5px, transparent 0.5px), radial-gradient(${color} 0.5px, transparent 0.5px)`,
                        backgroundSize: "20px 20px",
                        backgroundPosition: "0 0,10px 10px",
                    }}
                ></div>
            )}
            <div className="relative p-4">
                <div className="bg-neutral-100 dark:bg-neutral-900 -mt-12 rounded-2xl border-[6px] border-neutral-200 dark:border-black overflow-hidden w-fit">
                    <img src={icon} className="h-12 w-12 object-cover" />
                </div>
                <div className="my-2">
                    <div className="font-medium text-xl leading-none text-neutral-900 dark:text-neutral-100">
                        {name}
                    </div>
                    <div className="mt-2 text-sm text-neutral-600 dark:text-neutral-400 line-clamp-3">
                        {description}
                    </div>
                </div>
                <div className="flex flex-col items-center gap-2">
                    {links.map((value, index) => {
                        if (value.link) {
                            return (
                                <Link href={value.link} target="_blank" className="w-full">
                                    <button
                                        className={`rounded text-center w-full p-2 text-sm mt-2 hover:opacity-80 transition`}
                                        style={{
                                            backgroundColor: value.linkColor,
                                            color: value.textColor,
                                        }}
                                    >
                                        {value.linkText}
                                    </button>
                                </Link>
                            )
                        } else {
                            return (
                                <button
                                    className={`rounded text-center w-full p-2 text-sm mt-2 transition cursor-not-allowed`}
                                    disabled={true}
                                    style={{
                                        backgroundColor: value.linkColor,
                                        color: value.textColor,
                                    }}
                                >
                                    {value.linkText}
                                </button>
                            )
                        }
                    })}
                </div>
            </div>
        </div>
    );
}
